// ฟังก์ชันแสดงรูปภาพ
void displayImage() {
  OLED.clearBuffer(); // ล้างหน้าจอ

  if (isShowingFirstImage) {
OLED.firstPage();
  do {
  OLED.drawXBMP(0, 0, 128, 64, epd_blow);
  } while ( OLED.nextPage() );
  } else {
    OLED.firstPage();
  do {
  OLED.drawXBMP(0, 0, 128, 64, epd_buss);
  } while ( OLED.nextPage() );  }

  OLED.sendBuffer(); // อัปเดตหน้าจอ
}

void setup() {
  // เริ่มต้นการทำงานของหน้าจอ OLED และปุ่ม
  Wire.begin(SDA_OLED, SCL_OLED);
  OLED.begin();
  OLED.clearBuffer();
  OLED.setContrast(100);

  button.setDebounceTime(50); // กำหนดเวลา debounce ให้กับปุ่ม

  // แสดงรูปแรกเมื่อเริ่มต้น
  displayImage();
}

void loop() {
  button.loop(); // อัปเดตสถานะของปุ่ม

  // ตรวจสอบว่าปุ่มถูกกด
  if (button.isPressed()) {
    isShowingFirstImage = !isShowingFirstImage; // สลับสถานะของรูปภาพ
    displayImage(); // อัปเดตการแสดงผลของรูปภาพ
  }
}
